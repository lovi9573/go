package main

import (
  "fmt"
)

/*@author JWeaver; 
 */

type List struct {
	size int;
	head *Node;
	tail *Node;
}

type Node struct {
	prev *Node;
	next *Node;
	e interface{};
}

// SIMPLE LIST FUNCTIONS
func (l *List) addFront(a interface{}) interface{} {
  // O(1)
	temp := new (Node);
	temp.e=a;
	l.size+=1;

	if(l.head==l.tail&&l.head==nil){
		l.tail=temp; 
		l.head=temp;
		return a;
	}
	temp.next=l.head;
	l.head.prev=temp;
	l.head=temp;	
	return a;
}

func (l *List) removeFront() interface{}  {
  //O(1)
	if(l.size==0){
	   return nil;
	}
	retVal:=l.head.e;
	temp:=l.head.next;
	l.head.next=nil;
	l.head.prev=nil;
	l.size-=1;
	l.head=temp;
	return retVal;
}



func (l *List) addRear(a interface{}){
  //O(1)
	temp := new (Node);
	temp.e=a;
	l.size+=1;

	if(l.head==l.tail && l.tail==nil){
		l.tail=temp;
		l.head=temp;
		return;
	}
	temp.prev=l.tail;
	l.tail.next=temp;
	l.tail=temp;
}

func (l *List) removeRear() interface{} {
  //O(1)
	if(l.size==0){
           return nil;
        }
        retVal:=l.tail.e;
        temp:=l.tail.prev;
        l.tail.next=nil;
        l.tail.prev=nil;
        l.size-=1;
        l.tail=temp;
	return retVal;

}

func (l* List) removeAt(index int) interface{} {
  //O(n)
	if(index > l.size){return nil;}
	if(index == 0){	return l.removeFront();	}
	if(index==l.size-1){return l.removeRear();}

	temp:=l.head;
	for i:=0 ; i < index ; i+=1 {
		temp=temp.next;
	}
	retVal:=temp.e;
	temp.prev.next=temp.next;
	temp.next.prev=temp.prev;
	return retVal;
}

func (l* List) remove(e interface{}) interface{} {
  //O(n)
  if(l.head.e==e){ return l.removeFront();	}
  if(l.tail.e==e){ return l.removeRear();	}
    
  temp:=l.head.next;
  for temp.next != l.tail {
    if(temp.e == e){
      retVal:=temp.e;
	temp.prev.next=temp.next;
	temp.next.prev=temp.prev;
	l.size-=1;
	return retVal;
      }
      temp=temp.next;
    } 
    return nil;
}

func (l* List) clear() {
  //O(1) -- though not sure about garbage collection.
 l.head=nil;
 l.tail=nil;
 l.size=0;
}

func PrintList(n *Node)  {
  //O(n) -- recursive, possibly dangerous.
	if(n.next==nil){
		fmt.Printf("%v\n\n",n.e);
		return;
	}
	fmt.Printf("%v-",n.e);
	PrintList(n.next);
}

func (n *Node) toString() {
  //O(n) 
  if(n.next==nil){
    fmt.Printf("%v\n\n",n.e);
   return;    
  }
  fmt.Printf("%v-",n.e);
  n.next.toString();
}

func (l* List) Size() int {
 return l.size; 
}

//DRIVER
func main(){
	l:= new (List);
	for i:=0 ; i<10 ; i+=1 {
		l.addRear(i);	
	}
	fmt.Println("Initial list size: ",l.size);
	l.head.toString();
	
	fmt.Println("removed from front: ", l.removeFront(), " size: ",l.Size());
	l.head.toString();

	fmt.Println("removeAt(3), removed: ", l.removeAt(3), " size: ",l.Size());
	l.head.toString();
	
	fmt.Println("removeAt(0), removed: ", l.removeAt(0), " size: ",l.Size());
	l.head.toString();
	
	fmt.Println("removeAt(0), removed: ", l.removeAt(0), " size: ",l.Size());
	l.head.toString();
	
	fmt.Println("remove(7), removed: ", l.remove(7), " size: ",l.Size());
	l.head.toString();
	
	fmt.Println("remove(7), removed: ", l.remove(7), " size: ",l.Size());
	l.head.toString();
	
	fmt.Println("removeRear(), removed: ", l.removeRear(), " size: ",l.Size());
	l.head.toString();
	
	fmt.Println("addFront() removed: ", l.addFront("FOO"), " size: ",l.Size());
	l.head.toString();
}
